rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users collection
    match /users/{uid} {
      allow read, write: if request.auth.uid == uid;
    }

    // Requests collection
    match /requests/{reqId} {
      allow read: if request.auth != null;
      allow create: if request.auth.token.role == 'seller' || request.auth.token.role == 'admin';
      allow update, delete: if request.auth.token.role == 'seller' && resource.data.sellerId == request.auth.uid;
      
      // Bids subcollection
      match /bids/{bidId} {
        allow read: if request.auth != null;
        allow create: if request.auth.token.role == 'partner';
        allow update, delete: if request.auth.token.role == 'partner' && resource.data.partnerId == request.auth.uid;
      }
    }

    // Matches collection
    match /matches/{matchId} {
      allow read: if request.auth != null;
      allow write: if request.auth.token.role == 'admin';
    }

    // UserTokens collection
    match /userTokens/{uid} {
      allow read, write: if request.auth.uid == uid;
    }
  }
}
